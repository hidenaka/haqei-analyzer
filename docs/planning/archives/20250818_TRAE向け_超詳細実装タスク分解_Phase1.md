```javascript
/**
 * TabNavigator - OSアナライザー結果ページのタブ制御システム
 * 6つのタブ間の切り替えとコンテンツ表示を管理
 */
class TabNavigator {
    constructor() {
        this.tabs = [
            { id: 'basic', name: '基本', component: 'BasicResultsTab' },
            { id: 'transparency', name: '透明化', component: 'TransparencyTab' },
            { id: 'practical', name: '活用法', component: 'PracticalGuideTab' },
            { id: 'synergy', name: '統合', component: 'SynergyAnalysisTab' },
            { id: 'detailed', name: '詳細', component: 'DetailedTab' },
            { id: 'advanced', name: '専門', component: 'AdvancedTab' }
        ];
        
        this.currentTab = 'basic';
        this.tabComponents = new Map();
        this.containerElement = null;
        
        this.init();
    }
    
    /**
     * TabNavigator初期化
     */
    init() {
        this.createTabStructure();
        this.bindEvents();
        this.initializeComponents();
    }
    
    /**
     * タブのHTML構造を作成
     */
    createTabStructure() {
        const container = document.getElementById('virtual-persona-results-container');
        if (!container) {
            console.error('Results container not found');
            return;
        }
        
        this.containerElement = container;
        
        // タブナビゲーション作成
        const tabNavHTML = `
            <div class="tab-navigator">
                <div class="tab-header">
                    ${this.tabs.map(tab => `
                        <button class="tab-btn ${tab.id === 'basic' ? 'active' : ''}" 
                                data-tab="${tab.id}">
                            ${tab.name}
                        </button>
                    `).join('')}
                </div>
                <div class="tab-content-container">
                    ${this.tabs.map(tab => `
                        <div class="tab-content ${tab.id === 'basic' ? 'active' : ''}" 
                             data-tab-content="${tab.id}">
                            <div id="${tab.id}-tab-content"></div>
                        </div>
                    `).join('')}
                </div>
            </div>
        `;
        
        container.innerHTML = tabNavHTML;
    }
    
    /**
     * イベントリスナーの設定
     */
    bindEvents() {
        const tabButtons = document.querySelectorAll('.tab-btn');
        tabButtons.forEach(btn => {
            btn.addEventListener('click', (e) => {
                const tabId = e.target.dataset.tab;
                this.switchToTab(tabId);
            });
        });
    }
    
    /**
     * タブ切り替え処理
     */
    switchToTab(tabId) {
        if (this.currentTab === tabId) return;
        
        // 前のタブを非活性化
        this.deactivateTab(this.currentTab);
        
        // 新しいタブを活性化
        this.activateTab(tabId);
        
        this.currentTab = tabId;
        
        // アニメーション効果
        this.animateTabSwitch(tabId);
    }
    
    /**
     * タブの非活性化
     */
    deactivateTab(tabId) {
        const btn = document.querySelector(`[data-tab="${tabId}"]`);
        const content = document.querySelector(`[data-tab-content="${tabId}"]`);
        
        if (btn) btn.classList.remove('active');
        if (content) content.classList.remove('active');
    }
    
    /**
     * タブの活性化
     */
    activateTab(tabId) {
        const btn = document.querySelector(`[data-tab="${tabId}"]`);
        const content = document.querySelector(`[data-tab-content="${tabId}"]`);
        
        if (btn) btn.classList.add('active');
        if (content) content.classList.add('active');
        
        // タブコンテンツの更新
        this.updateTabContent(tabId);
    }
    
    /**
     * タブコンテンツの更新
     */
    updateTabContent(tabId) {
        const component = this.tabComponents.get(tabId);
        if (component && typeof component.render === 'function') {
            component.render();
        } else {
            // Phase 1では基本タブ以外は準備中メッセージ
            if (tabId !== 'basic') {
                this.renderComingSoonMessage(tabId);
            }
        }
    }
    
    /**
     * 準備中メッセージの表示
     */
    renderComingSoonMessage(tabId) {
        const container = document.getElementById(`${tabId}-tab-content`);
        if (container) {
            container.innerHTML = `
                <div class="coming-soon-message">
                    <div class="coming-soon-icon">🚧</div>
                    <h3>この機能は準備中です</h3>
                    <p>Phase 2で実装予定です。しばらくお待ちください。</p>
                </div>
            `;
        }
    }
    
    /**
     * タブ切り替えアニメーション
     */
    animateTabSwitch(tabId) {
        const content = document.querySelector(`[data-tab-content="${tabId}"]`);
        if (content) {
            content.style.opacity = '0';
            content.style.transform = 'translateY(10px)';
            
            setTimeout(() => {
                content.style.transition = 'all 0.3s ease';
                content.style.opacity = '1';
                content.style.transform = 'translateY(0)';
            }, 50);
        }
    }
    
    /**
     * コンポーネントの初期化
     */
    initializeComponents() {
        // Phase 1では基本タブのみ実装
        this.tabComponents.set('basic', null); // BasicResultsTabで後から設定
    }
    
    /**
     * タブコンポーネントの登録
     */
    registerTabComponent(tabId, component) {
        this.tabComponents.set(tabId, component);
    }
    
    /**
     * 現在のタブIDを取得
     */
    getCurrentTab() {
        return this.currentTab;
    }
    
    /**
     * 特定のタブが利用可能かチェック
     */
    isTabAvailable(tabId) {
        return this.tabComponents.has(tabId) && this.tabComponents.get(tabId) !== null;
    }
}

// グローバルインスタンス
```

```javascript
/**
 * BaseTabView - 全タブコンポーネントの基底クラス
 * 共通機能とインターフェースを定義
 */
class BaseTabView {
    constructor(tabId, tripleOSData) {
        this.tabId = tabId;
        this.tripleOSData = tripleOSData;
        this.container = null;
        this.isInitialized = false;
        
        this.initContainer();
    }
    
    /**
     * コンテナの初期化
     */
    initContainer() {
        this.container = document.getElementById(`${this.tabId}-tab-content`);
        if (!this.container) {
            console.error(`Container not found for tab: ${this.tabId}`);
        }
    }
    
    /**
     * レンダリング - サブクラスでオーバーライド必須
     */
    render() {
        throw new Error('render() method must be implemented by subclass');
    }
    
    /**
     * 初期化処理 - サブクラスでオーバーライド可能
     */
    initialize() {
        this.isInitialized = true;
    }
    
    /**
     * データ更新処理
     */
    updateData(newTripleOSData) {
        this.tripleOSData = newTripleOSData;
        if (this.isInitialized) {
            this.render();
        }
    }
    
    /**
     * 共通ユーティリティ: HTMLエスケープ
     */
    escapeHtml(text) {
        const div = document.createElement('div');
        div.textContent = text;
        return div.innerHTML;
    }
    
    /**
     * 共通ユーティリティ: 数値のフォーマット
     */
    formatScore(score) {
        return Math.round(score * 10) / 10;
    }
    
    /**
     * 共通ユーティリティ: 易卦名の取得
     */
    getHexagramName(hexagramData) {
        if (!hexagramData) return '未定義';
        return hexagramData.name || '未定義';
    }
    
    /**
     * 共通ユーティリティ: OSカードのHTML生成
     */
    generateOSCardHTML(osType, osData) {
        const osNames = {
            engine: 'Engine OS',
            interface: 'Interface OS',
            safemode: 'Safe Mode OS'
        };
        
        return `
            <div class="os-card ${osType}-os">
                <div class="os-header">
                    <h3>${osNames[osType]}</h3>
                    <div class="os-score">${this.formatScore(osData.score)}</div>
                </div>
                <div class="os-hexagram">
                    <div class="hexagram-symbol">${osData.hexagram.symbol}</div>
                    <div class="hexagram-name">${this.getHexagramName(osData.hexagram)}</div>
                </div>
                <div class="os-traits">
                    ${osData.traits.map(trait => `<span class="trait-tag">${trait}</span>`).join('')}
                </div>
            </div>
        `;
    }
    
    /**
     * エラー表示
     */
    showError(message) {
        if (this.container) {
            this.container.innerHTML = `
                <div class="error-message">
                    <div class="error-icon">⚠️</div>
                    <h3>エラーが発生しました</h3>
                    <p>${this.escapeHtml(message)}</p>
                </div>
            `;
        }
    }
    
    /**
     * ローディング表示
     */
    showLoading() {
        if (this.container) {
            this.container.innerHTML = `
                <div class="loading-message">
                    <div class="loading-spinner"></div>
                    <p>読み込み中...</p>
                </div>
            `;
        }
    }
    
    /**
     * クリーンアップ処理
     */
    destroy() {
        if (this.container) {
            this.container.innerHTML = '';
        }
        this.isInitialized = false;
    }
}

// グローバル公開
window.BaseTabView = BaseTabView;
```

```javascript
/**
 * BasicResultsTab - 基本結果表示タブ
 * Triple OSカードとスコアを表示する
 */
class BasicResultsTab extends BaseTabView {
    constructor(tripleOSData) {
        super('basic', tripleOSData);
        this.initialize();
    }
    
    /**
     * 初期化処理
     */
    initialize() {
        super.initialize();
        this.render();
    }
    
    /**
     * メインレンダリング
     */
    render() {
        if (!this.container) {
            this.showError('表示コンテナが見つかりません');
            return;
        }
        
        if (!this.tripleOSData) {
            this.showError('Triple OSデータが設定されていません');
            return;
        }
        
        try {
            const html = this.generateBasicResultsHTML();
            this.container.innerHTML = html;
            this.bindEvents();
        } catch (error) {
            console.error('BasicResultsTab render error:', error);
            this.showError('結果の表示中にエラーが発生しました');
        }
    }
    
    /**
     * 基本結果のHTML生成
     */
    generateBasicResultsHTML() {
        return `
            <div class="basic-results-content">
                ${this.generateHeaderSection()}
                ${this.generateTripleOSSection()}
                ${this.generateSummarySection()}
                ${this.generateActionButtons()}
            </div>
        `;
    }
    
    /**
     * ヘッダーセクション生成
     */
    generateHeaderSection() {
        return `
            <div class="results-header">
                <h2>🎯 あなたのTriple OS分析結果</h2>
                <p class="results-subtitle">
                    36の質問から導き出された、あなたの内なる3つのOSの特性です
                </p>
                <div class="analysis-date">
                    分析日時: ${new Date().toLocaleDateString('ja-JP')}
                </div>
            </div>
        `;
    }
    
    /**
     * Triple OSセクション生成
     */
    generateTripleOSSection() {
        const { engineOS, interfaceOS, safeModeOS } = this.tripleOSData;
        
        return `
            <div class="triple-os-container">
                <div class="os-cards-grid">
                    ${this.generateOSCardHTML('engine', engineOS)}
                    ${this.generateOSCardHTML('interface', interfaceOS)}
                    ${this.generateOSCardHTML('safemode', safeModeOS)}
                </div>
                ${this.generateOSRelationship()}
            </div>
        `;
    }
    
    /**
     * OS間の関係性表示
     */
    generateOSRelationship() {
        return `
            <div class="os-relationship">
                <h3>🔄 3つのOSの関係性</h3>
                <div class="relationship-diagram">
                    <div class="relationship-line engine-interface">
                        <span class="relationship-label">表現の一貫性</span>
                    </div>
                    <div class="relationship-line engine-safe">
                        <span class="relationship-label">内的統合</span>
                    </div>
                    <div class="relationship-line interface-safe">
                        <span class="relationship-label">安定性バランス</span>
                    </div>
                </div>
                <p class="relationship-description">
                    Engine OSが内なる価値観、Interface OSが外向きの表現、
                    Safe Mode OSが安心・安全の基盤として機能しています。
                </p>
            </div>
        `;
    }
    
    /**
     * サマリーセクション生成
     */
    generateSummarySection() {
        const overallPattern = this.analyzeOverallPattern();
        
        return `
            <div class="results-summary">
                <h3>📊 総合分析サマリー</h3>
                <div class="summary-cards">
                    <div class="summary-card personality-type">
                        <h4>人格タイプ</h4>
                        <p>${overallPattern.personalityType}</p>
                    </div>
                    <div class="summary-card strength-area">
                        <h4>強みエリア</h4>
                        <p>${overallPattern.strengthArea}</p>
                    </div>
                    <div class="summary-card growth-potential">
                        <h4>成長ポテンシャル</h4>
                        <p>${overallPattern.growthPotential}</p>
                    </div>
                </div>
                <div class="key-insights">
                    <h4>🔍 重要な洞察</h4>
                    <ul>
                        ${overallPattern.keyInsights.map(insight => 
                            `<li>${insight}</li>`
                        ).join('')}
                    </ul>
                </div>
            </div>
        `;
    }
    
    /**
     * アクションボタン生成
     */
    generateActionButtons() {
        return `
            <div class="action-buttons">
                <button class="btn btn-primary explore-transparency">
                    🔍 分析過程を見る（透明化タブ）
                </button>
                <button class="btn btn-secondary explore-practical">
                    🎯 活用法を見る（活用法タブ）
                </button>
                <button class="btn btn-accent explore-synergy">
                    ⚡ シナジー分析を見る（統合タブ）
                </button>
                <button class="btn btn-outline save-results">
                    💾 結果を保存
                </button>
            </div>
        `;
    }
    
    /**
     * 全体パターンの分析
     */
    analyzeOverallPattern() {
        const { engineOS, interfaceOS, safeModeOS } = this.tripleOSData;
        
        // スコアバランスの分析
        const scores = [engineOS.score, interfaceOS.score, safeModeOS.score];
        const maxScore = Math.max(...scores);
        const minScore = Math.min(...scores);
        const scoreRange = maxScore - minScore;
        
        // 主導OSの決定
        let dominantOS = 'engine';
        if (interfaceOS.score > engineOS.score && interfaceOS.score > safeModeOS.score) {
            dominantOS = 'interface';
        } else if (safeModeOS.score > engineOS.score && safeModeOS.score > interfaceOS.score) {
            dominantOS = 'safemode';
        }
        
        // パターン分析
        let personalityType = '';
        let strengthArea = '';
        let growthPotential = '';
        let keyInsights = [];
        
        if (scoreRange < 20) {
            personalityType = 'バランス型人格';
            strengthArea = '全方位的な適応力';
            growthPotential = '状況に応じた柔軟な対応';
            keyInsights = [
                '3つのOSがバランスよく発達しています',
                '様々な場面で安定したパフォーマンスを発揮できます',
                '特定の状況に特化した能力開発が成長のカギです'
            ];
        } else if (dominantOS === 'engine') {
            personalityType = 'イノベーター型';
            strengthArea = '創造性とリーダーシップ';
            growthPotential = 'アイデアの実現力向上';
            keyInsights = [
                'Engine OSが強く、革新的な思考を持っています',
                'Interface OSの強化で影響力を拡大できます',
                'Safe Mode OSの安定性も重要な成長要素です'
            ];
        } else if (dominantOS === 'interface') {
            personalityType = 'コミュニケーター型';
            strengthArea = '対人関係と表現力';
            growthPotential = '深い洞察力の獲得';
            keyInsights = [
                'Interface OSが優秀で、人との関係構築が得意です',
                'Engine OSの強化で独自性を高められます',
                'Safe Mode OSで心の安定を保つことが重要です'
            ];
        } else {
            personalityType = 'スタビライザー型';
            strengthArea = '安定性と信頼性';
            growthPotential = '積極性と表現力の向上';
            keyInsights = [
                'Safe Mode OSが強く、安定した基盤を持っています',
                'Engine OSの活性化で新しい挑戦ができます',
                'Interface OSで自分の価値をより伝えられます'
            ];
        }
        
        return {
            personalityType,
            strengthArea,
            growthPotential,
            keyInsights
        };
    }
    
    /**
     * イベントバインディング
     */
    bindEvents() {
        // 透明化タブへの遷移
        const transparencyBtn = this.container.querySelector('.explore-transparency');
        if (transparencyBtn) {
            transparencyBtn.addEventListener('click', () => {
                if (window.tabNavigator) {
                    window.tabNavigator.switchToTab('transparency');
                }
            });
        }
        
        // 活用法タブへの遷移
        const practicalBtn = this.container.querySelector('.explore-practical');
        if (practicalBtn) {
            practicalBtn.addEventListener('click', () => {
                if (window.tabNavigator) {
                    window.tabNavigator.switchToTab('practical');
                }
            });
        }
        
        // 統合タブへの遷移
        const synergyBtn = this.container.querySelector('.explore-synergy');
        if (synergyBtn) {
            synergyBtn.addEventListener('click', () => {
                if (window.tabNavigator) {
                    window.tabNavigator.switchToTab('synergy');
                }
            });
        }
        
        // 結果保存
        const saveBtn = this.container.querySelector('.save-results');
        if (saveBtn) {
            saveBtn.addEventListener('click', () => {
                this.saveResults();
            });
        }
        
        // OSカードのホバーエフェクト
        const osCards = this.container.querySelectorAll('.os-card');
        osCards.forEach(card => {
            card.addEventListener('mouseenter', (e) => {
                e.target.classList.add('hovered');
            });
            
            card.addEventListener('mouseleave', (e) => {
                e.target.classList.remove('hovered');
            });
        });
    }
    
    /**
     * 結果保存機能
     */
    saveResults() {
        try {
            const resultsData = {
                timestamp: new Date().toISOString(),
                tripleOS: this.tripleOSData,
                analysis: this.analyzeOverallPattern()
            };
            
            localStorage.setItem('haqei_os_analysis_results', JSON.stringify(resultsData));
            
            // 成功メッセージ表示
            this.showSuccessMessage('結果が保存されました');
            
        } catch (error) {
            console.error('Save results error:', error);
            this.showErrorMessage('結果の保存に失敗しました');
        }
    }
    
    /**
     * 成功メッセージ表示
     */
    showSuccessMessage(message) {
        const toast = document.createElement('div');
        toast.className = 'toast toast-success';
        toast.textContent = message;
        document.body.appendChild(toast);
        
        setTimeout(() => {
            toast.remove();
        }, 3000);
    }
    
    /**
     * エラーメッセージ表示
     */
    showErrorMessage(message) {
        const toast = document.createElement('div');
        toast.className = 'toast toast-error';
        toast.textContent = message;
        document.body.appendChild(toast);
        
        setTimeout(() => {
            toast.remove();
        }, 3000);
    }
}
```
