/**
 * バランスタイプをわかりやすく説明
 */
determineBalanceType(scores) {
    const max = Math.max(...scores);
    const min = Math.min(...scores);
    const diff = max - min;
    
    // OSの日本語名
    const osNames = ['内なる自分', '社交的な自分', '守りの自分'];
    const maxIndex = scores.indexOf(max);
    const minIndex = scores.indexOf(min);
    
    if (diff < 10) {
        return {
            className: 'balanced',
            icon: '⚖️',
            title: 'バランス型',
            description: 'あなたは状況に応じて、うまく自分を切り替えられるタイプです。内面も、人付き合いも、守りの姿勢も、どれもバランスよく持っています。',

            dailyExample: '仕事では積極的に、家では穏やかに、ストレス時は冷静に対処できています。'
