        'ベテラン', '最終判断', '裁定', '決裁', '承認',
        '認可', '批准', '決着', '重鎮', '要職',

    ];
    
    // マッチング処理
    for (const keyword of extendedKeywords) {
        if (text.includes(keyword)) {
            bonus += 0.25; // 従来の0.1から0.25に増加
            break;
        }
    }
}
